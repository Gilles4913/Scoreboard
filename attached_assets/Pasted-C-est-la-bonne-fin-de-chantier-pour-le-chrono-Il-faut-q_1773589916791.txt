C’est la bonne fin de chantier pour le chrono.

Il faut que les 3 surfaces appliquent exactement la même règle :

émettre live_seq

mémoriser lastAppliedSeq

ignorer tout patch plus ancien

recaler l’horloge uniquement depuis une ancre valide

Voici ce qu’il faut faire sur la console principale dans apps/operator/src/pages/ControlPage.tsx.

1) Ajouter les refs live_seq

Ajoute avec les autres useRef(...) :

const liveSeqRef = useRef<number>(0);
const lastAppliedSeqRef = useRef<number>(0);

function nextLiveSeq() {
  liveSeqRef.current += 1;
  return liveSeqRef.current;
}
2) Ajouter un helper pour appliquer un patch entrant

Ajoute cette fonction dans le composant :

function applyIncomingLivePatch(patch: Record<string, any>) {
  if (!patch || typeof patch !== "object") return;

  const seq = Number(patch.live_seq || 0);

  if (seq && seq < lastAppliedSeqRef.current) {
    return;
  }

  if (seq) {
    lastAppliedSeqRef.current = seq;
    if (seq > liveSeqRef.current) {
      liveSeqRef.current = seq;
    }
  }

  if (typeof patch.status === "string") {
    setStatus(patch.status);
  }

  if (typeof patch.period_label === "string") {
    setPeriodLabel(patch.period_label);
  }

  if (typeof patch.home_score === "number") {
    setHomeScore(patch.home_score);
  }

  if (typeof patch.away_score === "number") {
    setAwayScore(patch.away_score);
  }

  if (typeof patch.home_team_fouls === "number") {
    setHomeTeamFouls(patch.home_team_fouls);
  }

  if (typeof patch.away_team_fouls === "number") {
    setAwayTeamFouls(patch.away_team_fouls);
  }

  if (typeof patch.home_timeouts === "number") {
    setHomeTimeouts(patch.home_timeouts);
  }

  if (typeof patch.away_timeouts === "number") {
    setAwayTimeouts(patch.away_timeouts);
  }

  if (typeof patch.home_bonus === "boolean") {
    setHomeBonus(patch.home_bonus);
  }

  if (typeof patch.away_bonus === "boolean") {
    setAwayBonus(patch.away_bonus);
  }

  if (typeof patch.shot_clock_s === "number") {
    setShotClockS(patch.shot_clock_s);
  }

  if (typeof patch.possession_arrow === "string" || patch.possession_arrow === null) {
    setPossessionArrow(patch.possession_arrow);
  }

  const hasAnchorEpoch = typeof patch.clock_anchor_epoch === "number";
  const hasAnchorMs = typeof patch.clock_anchor_ms === "number";
  const hasClockMs = typeof patch.clock_ms === "number";
  const hasClockRunning = typeof patch.clock_running === "boolean";

  if (hasAnchorEpoch && hasAnchorMs) {
    clockAnchorRef.current = {
      epoch: patch.clock_anchor_epoch,
      ms: patch.clock_anchor_ms,
    };
  }

  if (hasClockMs) {
    clockMsRef.current = patch.clock_ms;
    setClockMs(patch.clock_ms);
  }

  if (hasClockRunning) {
    clockRunningRef.current = !!patch.clock_running;
    setClockRunning(!!patch.clock_running);
  }
}

4) Faire suivre last_event_seq à la persistance DB

Dans toutes les actions critiques (startClock, pauseClock, resetClock, changeScore, etc.), quand tu fais persistLiveState(...), passe aussi last_event_seq.

Exemple pour pauseClock()

Remplace entièrement pauseClock() par ceci :

async function pauseClock() {
  const nextClockMs = clockMs;
  const now = Date.now();
  const seq = nextLiveSeq();

  clockAnchorRef.current = { epoch: now, ms: nextClockMs };
  setClockRunning(false);
  setStatus("paused");

  const patch = {
    clock_running: false,
    clock_ms: nextClockMs,
    status: "paused",
    live_seq: seq,
    clock_anchor_epoch: now,
    clock_anchor_ms: nextClockMs,
    emitted_at: now,
  };

  lastAppliedSeqRef.current = seq;

  try {
    await sendTvBroadcast(match!.id, {
      match_id: match!.id,
      ...patch,
    });

    void persistLiveState({
      clock_running: false,
      clock_ms: nextClockMs,
      status: "paused",
      last_event_seq: seq,
    });
  } catch {}
}
Exemple pour startClock()

Remplace entièrement startClock() par ceci :

async function startClock() {
  const nextClockMs =
    clockMs > 0 ? clockMs : defaultClockMsBySport(sport, sportSettings?.period_duration_s);

  const now = Date.now();
  const seq = nextLiveSeq();

  clockAnchorRef.current = { epoch: now, ms: nextClockMs };
  setClockMs(nextClockMs);
  setClockRunning(true);
  setStatus("live");

  lastAppliedSeqRef.current = seq;

  try {
    await sendTvBroadcast(match!.id, {
      match_id: match!.id,
      clock_ms: nextClockMs,
      clock_running: true,
      status: "live",
      live_seq: seq,
      clock_anchor_epoch: now,
      clock_anchor_ms: nextClockMs,
      emitted_at: now,
    });

    void persistLiveState({
      clock_ms: nextClockMs,
      clock_running: true,
      status: "live",
      last_event_seq: seq,
    });
  } catch {}
}

5) Si ControlPage reçoit des patches entrants, filtrer aussi par live_seq

Si tu as un channel.on("broadcast"... ou un postgres_changes dans ControlPage.tsx, fais passer les messages par applyIncomingLivePatch(...).

Broadcast
.on("broadcast", { event: "*" }, (message) => {
  const patch = message?.payload?.patch || message?.payload || message;
  if (!patch || typeof patch !== "object") return;
  applyIncomingLivePatch(patch);
})
Postgres changes
.on(
  "postgres_changes",
  {
    event: "UPDATE",
    schema: "public",
    table: "matches",
    filter: `id=eq.${match.id}`,
  },
  (payload: any) => {
    const row = payload?.new;
    if (!row) return;

    applyIncomingLivePatch({
      live_seq: row.last_event_seq ?? 0,
      status: row.status,
      period_label: row.period_label,
      home_score: row.home_score,
      away_score: row.away_score,
