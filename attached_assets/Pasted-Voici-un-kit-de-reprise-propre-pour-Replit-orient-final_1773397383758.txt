Voici un kit de reprise propre pour Replit, orienté finalisation pragmatique de scoreDisplay V2.

Le projet existe déjà, avec Supabase, un Operator multi-sport, un Display multi-sport, des équipes, joueurs, feuilles de match, réglages Display et réglages sport. La direction validée est le Display public stable par équipe via teamSlug, avec abandon progressif des anciens reliquats token match. Le projet est en phase de stabilisation / alignement code-DB / durcissement produit.

Migration B — bibliothèque de templates d’affichage
begin;

create table if not exists public.display_templates (
  id uuid primary key default gen_random_uuid(),
  code text not null unique,
  name text not null,
  sport text not null,
  layout_mode text not null,
  config_json jsonb not null default '{}'::jsonb,
  is_default_system boolean not null default false,
  is_active boolean not null default true,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists public.team_display_settings (
  team_id uuid primary key references public.teams(id) on delete cascade,
  default_display_template_id uuid references public.display_templates(id) on delete set null,
  theme_override text null,
  show_logos_override boolean null,
  show_sponsors_override boolean null,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create index if not exists idx_display_templates_sport on public.display_templates(sport);
create index if not exists idx_display_templates_active on public.display_templates(is_active);
create index if not exists idx_team_display_settings_template on public.team_display_settings(default_display_template_id);

  ('volley_sets', 'Volley Sets', 'volleyball', 'volley_sets', '{"density":"medium"}', true, true)
on conflict (code) do nothing;

insert into public.team_display_settings (team_id, default_display_template_id)
select
  t.id,
  dt.id
from public.teams t
join public.orgs o on o.id = t.org_id
join public.display_templates dt
  on dt.sport = o.sport
 and dt.is_default_system = true
where not exists (
  select 1
  from public.team_display_settings s
  where s.team_id = t.id
);
